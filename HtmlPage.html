Step one - description
This example shows how to integrate Dynamic Step Process Panels with form.

Features:
- 3 steps (width configurable in css),
- custom form validation,
- first step selected,
- step-by-step mode,
- previous and next buttons,
- step labels,
- every step can be choosed again,
- fade effect speed 1 second,
- style orange-blue

$("#processPanel9").processPanel({
	style: "blue-orange",
	onOpen: function(event, step, content, stepNumber){
		if(stepNumber==2)
		{
			$("#p9content1 .alert").remove();
			if($("[name=name]").val()=="")
			{
				$("#p9content1").prepend("<div class='alert'>Please enter 
				your name!</div>");
				return false;
			}
		}
		else if(stepNumber==3)
		{
			$("#p9content2 .alert").remove();
			if(!checkEmail($("[name=email]").val()))
			{
				$("#p9content2").prepend("<div class='alert'>Please enter
				valid e-mail address!</div>");
				return false;
			}
		}
	}
});
